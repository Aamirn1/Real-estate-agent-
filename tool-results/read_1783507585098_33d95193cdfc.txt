   380→   380→
   381→   381→function CrmKanban() {
   382→   382→  const [columns, setColumns] = useState<Column[]>(INITIAL_COLUMNS);
   383→   383→  const [activeLead, setActiveLead] = useState<Lead | null>(null);
   384→   384→
   385→   385→  const sensors = useSensors(
   386→   386→    useSensor(PointerSensor, { activationConstraint: { distance: 6 } })
   387→   387→  );
   388→   388→
   389→   389→  function handleDragStart(event: DragStartEvent) {
   390→   390→    const id = String(event.active.id);
   391→   391→    const lead = columns
   392→   392→      .flatMap((c) => c.leads)
   393→   393→      .find((l) => l.id === id);
   394→   394→    setActiveLead(lead ?? null);
   395→   395→  }
   396→   396→
   397→   397→  function handleDragEnd(event: DragEndEvent) {
   398→   398→    setActiveLead(null);
   399→   399→    const { active, over } = event;
   400→   400→    if (!over) return;
   401→   401→
   402→   402→    const activeId = String(active.id);
   403→   403→    const overId = String(over.id);
   404→   404→    if (activeId === overId) return;
   405→   405→
   406→   406→    setColumns((prev) => {
   407→   407→      const srcCol = prev.find((c) => c.leads.some((l) => l.id === activeId));
   408→   408→      if (!srcCol) return prev;
   409→   409→      const activeLeadCard = srcCol.leads.find((l) => l.id === activeId);
   410→   410→      if (!activeLeadCard) return prev;
   411→   411→
   412→   412→      // Dropped onto a column droppable (empty area / end of column)
   413→   413→      const destColById = prev.find((c) => c.id === overId);
   414→   414→      if (destColById) {
   415→   415→        if (destColById.id === srcCol.id) return prev;
   416→   416→        return prev.map((c) => {
   417→   417→          if (c.id === srcCol.id)
   418→   418→            return { ...c, leads: c.leads.filter((l) => l.id !== activeId) };
   419→   419→          if (c.id === destColById.id)
   420→   420→            return { ...c, leads: [...c.leads, activeLeadCard] };
   421→   421→          return c;
   422→   422→        });
   423→   423→      }
   424→   424→
   425→   425→      // Dropped onto another card
   426→   426→      const overCol = prev.find((c) => c.leads.some((l) => l.id === overId));
   427→   427→      if (!overCol) return prev;
   428→   428→
   429→   429→      if (overCol.id === srcCol.id) {
   430→   430→        const oldIndex = srcCol.leads.findIndex((l) => l.id === activeId);
   431→   431→        const newIndex = srcCol.leads.findIndex((l) => l.id === overId);
   432→   432→        if (oldIndex === -1 || newIndex === -1 || oldIndex === newIndex)
   433→   433→          return prev;
   434→   434→        return prev.map((c) =>
   435→   435→          c.id === srcCol.id
   436→   436→            ? { ...c, leads: arrayMove(c.leads, oldIndex, newIndex) }
   437→   437→            : c
   438→   438→        );
   439→   439→      }
   440→   440→
   441→   441→      // Cross-column move to a specific index
   442→   442→      const overIndex = overCol.leads.findIndex((l) => l.id === overId);
   443→   443→      return prev.map((c) => {
   444→   444→        if (c.id === srcCol.id)
   445→   445→          return { ...c, leads: c.leads.filter((l) => l.id !== activeId) };
   446→   446→        if (c.id === overCol.id) {
   447→   447→          const next = [...c.leads];
   448→   448→          next.splice(overIndex, 0, activeLeadCard);
   449→   449→          return { ...c, leads: next };
   450→   450→        }
   451→   451→        return c;
   452→   452→      });
   453→   453→    });
   454→   454→  }
   455→   455→
   456→   456→  return (
   457→   457→    <DndContext
   458→   458→      sensors={sensors}
   459→   459→      collisionDetection={closestCorners}
   460→   460→      onDragStart={handleDragStart}
   461→   461→      onDragEnd={handleDragEnd}
   462→   462→    >
   463→   463→      <div className="custom-scroll -mx-1 overflow-x-auto px-1 pb-3">
   464→   464→        <div className="flex items-start gap-3">
   465→   465→          {columns.map((col) => (
   466→   466→            <KanbanColumn key={col.id} column={col} />
   467→   467→          ))}
   468→   468→        </div>
   469→   469→      </div>
   470→   470→
   471→   471→      <DragOverlay dropAnimation={{ duration: 220, easing: "cubic-bezier(0.22,1,0.36,1)" }}>
   472→   472→        {activeLead ? (
   473→   473→          <div className="rotate-2 scale-[1.03] opacity-95">
   474→   474→            <LeadCardStatic lead={activeLead} />
   475→   475→          </div>
   476→   476→        ) : null}
   477→   477→      </DragOverlay>
   478→   478→    </DndContext>
   479→   479→  );
   480→   480→}

   667→   667→function TestimonialsCarousel() {
   668→   668→  const [api, setApi] = useState<CarouselApi | null>(null);
   669→   669→  const [current, setCurrent] = useState(0);
   670→   670→  const [count, setCount] = useState(0);
   671→   671→  const [paused, setPaused] = useState(false);
   672→   672→
   673→   673→  useEffect(() => {
   674→   674→    if (!api) return;
   675→   675→    const onSelect = () => {
   676→   676→      setCount(api.scrollSnapList().length);
   677→   677→      setCurrent(api.selectedScrollSnap());
   678→   678→    };
   679→   679→    api.on("select", onSelect);
   680→   680→    api.on("reInit", onSelect);
   681→   681→    // Defer initial sync to avoid synchronous setState in the effect body.
   682→   682→    queueMicrotask(onSelect);
   683→   683→    return () => {
   684→   684→      api.off("select", onSelect);
   685→   685→      api.off("reInit", onSelect);
   686→   686→    };
   687→   687→  }, [api]);
   688→   688→
   689→   689→  useEffect(() => {
   690→   690→    if (!api || paused) return;
   691→   691→    const id = setInterval(() => {
   692→   692→      api.scrollNext();
   693→   693→    }, 5000);
   694→   694→    return () => clearInterval(id);
   695→   695→  }, [api, paused]);
   696→   696→
   697→   697→  return (
   698→   698→    <div
   699→   699→      className="relative mx-auto max-w-3xl"
   700→   700→      onMouseEnter={() => setPaused(true)}
   701→   701→      onMouseLeave={() => setPaused(false)}
   702→   702→    >
   703→   703→      <Carousel
   704→   704→        opts={{ loop: true, align: "center" }}
   705→   705→        setApi={setApi}
   706→   706→        className="overflow-visible"
   707→   707→      >
   708→   708→        <CarouselContent className="-ml-4">
   709→   709→          {TESTIMONIALS.map((t, i) => (
   710→   710→            <CarouselItem key={t.name} className="basis-full pl-4">
   711→   711→              <TestimonialCard t={t} />
   712→   712→            </CarouselItem>
   713→   713→          ))}
   714→   714→        </CarouselContent>
   715→   715→
   716→   716→        <CarouselPrevious className="left-0 top-1/2 hidden h-10 w-10 -translate-y-1/2 border-white/15 bg-black/40 text-white/70 backdrop-blur-md hover:bg-white/10 hover:text-white sm:flex" />
   717→   717→        <CarouselNext className="right-0 top-1/2 hidden h-10 w-10 -translate-y-1/2 border-white/15 bg-black/40 text-white/70 backdrop-blur-md hover:bg-white/10 hover:text-white sm:flex" />
   718→   718→      </Carousel>
   719→   719→
   720→   720→      {/* dots */}
   721→   721→      <div className="mt-7 flex items-center justify-center gap-2">
   722→   722→        {Array.from({ length: count || TESTIMONIALS.length }).map((_, i) => (
   723→   723→          <button
   724→   724→            key={i}
   725→   725→            type="button"
   726→   726→            aria-label={`Go to testimonial ${i + 1}`}
   727→   727→            onClick={() => api?.scrollTo(i)}
   728→   728→            className="h-1.5 rounded-full transition-all duration-300"
   729→   729→            style={{
   730→   730→              width: i === current ? 26 : 8,
   731→   731→              backgroundColor:
   732→   732→                i === current ? "#3b82f6" : "rgba(255,255,255,0.2)",
   733→   733→              boxShadow:
   734→   734→                i === current ? "0 0 12px rgba(59,130,246,0.6)" : "none",
   735→   735→            }}
   736→   736→          />
   737→   737→        ))}
   738→   738→      </div>
   739→   739→    </div>
